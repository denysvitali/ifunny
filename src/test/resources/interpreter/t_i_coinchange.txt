
{coin change ->
    coin = {(index) ->
        if index == 0
        then 500
        else if index == 1
            then 200
            else if index == 2
                then 100
                else if index == 3
                    then 50
                    else if index == 4
                        then 20
                        else if index == 5
                            then 10
                            else if index == 6
                                then 5
                                fi
                            fi
                        fi
                    fi
                fi
            fi
        fi
    };
    
    change = {(amount index) ->
        if amount == 0 then 1
        else if amount < 0 then 0
             else if index >= 7 then 0
                  else change(amount, index + 1) + change(amount - coin(index), index)
                  fi
             fi
        fi
    };
    
    println(change(50, 0))
}
